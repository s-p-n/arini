$temp = {
	$types: [%type%],
	$values: {}
};
$val = %value%;
for ($i = 0; $i < $temp.$types.length; $i += 1) {
	if ($val.$values[%type%] !== void 0) {
		$temp.$values[%type%] = $val.$values[%type%];
	} else if ($val.$values["Scope"] !== void 0){
		$temp.$values[%type%] = function () {
			return $val.$values["Scope"]()().$values[%type%]();
		}
	}
}
